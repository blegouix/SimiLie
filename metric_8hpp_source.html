171template <class Indices1, class Indices2>
172struct FillMetricProd;
173
174template <>
175struct FillMetricProd<ddc::detail::TypeSeq<>, ddc::detail::TypeSeq<>>
176{
177 template <class MetricProdType, class MetricType, class MetricProdType_>
178 static MetricProdType run(
179 MetricProdType metric_prod,
180 MetricType metric,
181 MetricProdType_ metric_prod_)
182 {
183 Kokkos::deep_copy(
184 metric_prod.allocation_kokkos_view(),
185 metric_prod_
186 .allocation_kokkos_view()); // We rely on Kokkos::deep_copy in place of ddc::parallel_deepcopy to avoid type verification of the type dimensions
187 return metric_prod;
188 }
189};

191template <class HeadIndex1, class... TailIndex1, class HeadIndex2, class... TailIndex2>
192struct FillMetricProd<
193 ddc::detail::TypeSeq<HeadIndex1, TailIndex1...>,
194 ddc::detail::TypeSeq<HeadIndex2, TailIndex2...>>
195{
196 template <class MetricProdType, class MetricType, class MetricProdType_>
197 static MetricProdType run(
198 MetricProdType metric_prod,
199 MetricType metric,
200 MetricProdType_ metric_prod_)
201 {
202 ddc::cartesian_prod_t<
203 typename MetricProdType_::discrete_domain_type,
204 relabelize_metric_in_domain_t<
205 typename MetricType::indices_domain_t,
206 HeadIndex1,
207 HeadIndex2>>
208 new_metric_prod_dom_(
209 metric_prod_.domain(),
210 relabelize_metric_in_domain<HeadIndex1, HeadIndex2>(
211 metric.indices_domain()));
212 ddc::Chunk new_metric_prod_alloc_(new_metric_prod_dom_, ddc::HostAllocator<double>());
213 tensor::Tensor<
214 double,
215 ddc::cartesian_prod_t<
216 typename MetricProdType_::discrete_domain_type,
217 relabelize_metric_in_domain_t<
218 typename MetricType::indices_domain_t,
219 HeadIndex1,
220 HeadIndex2>>,
221 Kokkos::layout_right,
222 Kokkos::DefaultHostExecutionSpace::memory_space>
223 new_metric_prod_(new_metric_prod_alloc_);
224
225 if (new_metric_prod_dom_.size() != 0) {
226 ddc::for_each(new_metric_prod_.non_indices_domain(), [&](auto elem) {
227 tensor_prod(
228 new_metric_prod_[elem],
229 metric_prod_[elem],
230 relabelize_metric<HeadIndex1, HeadIndex2>(metric)[elem]);
231 });
232 }
233
234
235 return FillMetricProd<
236 ddc::detail::TypeSeq<TailIndex1...>,
237 ddc::detail::TypeSeq<TailIndex2...>>::run(metric_prod, metric, new_metric_prod_);
238 }
239};
240} // namespace detail

273// Apply metrics inplace (like g_mu_muprime*T^muprime^nu)
274template <misc::Specialization<Tensor> MetricType, misc::Specialization<Tensor> TensorType>
275auto inplace_apply_metric(TensorType tensor, MetricType metric_prod)
276{
277 ddc::Chunk result_alloc(
278 relabelize_indices_in<
279 swap_character<
280 detail::non_primes<typename MetricType::accessor_t::natural_domain_t>>,
281 detail::non_primes<typename MetricType::accessor_t::natural_domain_t>>(
282 tensor.domain()),
283 ddc::HostAllocator<double>());
284 relabelize_indices_of_t<
285 TensorType,
286 swap_character<detail::non_primes<typename MetricType::accessor_t::natural_domain_t>>,
287 detail::non_primes<typename MetricType::accessor_t::natural_domain_t>>
288 result(result_alloc);
289 ddc::for_each( // TODO use parallel_for_each, there is a weird lock when doing so
290 tensor.non_indices_domain(),
291 [&](auto elem) {
292 tensor_prod(
293 result[elem],
294 metric_prod[elem],
295 relabelize_indices_of<
296 swap_character<detail::non_primes<
297 typename MetricType::accessor_t::natural_domain_t>>,
298 primes<swap_character<detail::non_primes<
299 typename MetricType::accessor_t::natural_domain_t>>>>(
300 tensor)[elem]);
301 });
302 Kokkos::deep_copy(
303 tensor.allocation_kokkos_view(),
304 result.allocation_kokkos_view()); // We rely on Kokkos::deep_copy in place of ddc::parallel_deepcopy to avoid type verification of the type dimensions
305
306 return relabelize_indices_of<
307 swap_character<detail::non_primes<typename MetricType::accessor_t::natural_domain_t>>,
308 detail::non_primes<typename MetricType::accessor_t::natural_domain_t>>(tensor);
309}
