95template <>
96struct FillMetricProd<ddc::detail::TypeSeq<>, ddc::detail::TypeSeq<>>
97{
98 template <class MetricType, class MetricProdType, class MetricProdType_>
99 static MetricProdType run(
100 MetricProdType metric_prod,
101 MetricType metric,
102 MetricProdType_ metric_prod_)
103 {
104 Kokkos::deep_copy(
105 metric_prod.allocation_kokkos_view(),
106 metric_prod_
107 .allocation_kokkos_view()); // We rely on Kokkos::deep_copy in place of ddc::parallel_deepcopy to avoid type verification of the type dimensions
108 return metric_prod;
109 }
110};

112template <class HeadIndex1, class... TailIndex1, class HeadIndex2, class... TailIndex2>
113struct FillMetricProd<
114 ddc::detail::TypeSeq<HeadIndex1, TailIndex1...>,
115 ddc::detail::TypeSeq<HeadIndex2, TailIndex2...>>
116{
117 template <class MetricType, class MetricProdType, class MetricProdType_>
118 static MetricProdType run(
119 MetricProdType metric_prod,
120 MetricType metric,
121 MetricProdType_ metric_prod_)
122 {
123 ddc::cartesian_prod_t<
124 typename MetricProdType_::discrete_domain_type,
125 relabelize_metric_in_domain_t<
126 typename MetricType::discrete_domain_type,
127 HeadIndex1,
128 HeadIndex2>>
129 new_metric_prod_dom_(
130 metric_prod_.domain(),
131 relabelize_metric_in_domain<HeadIndex1, HeadIndex2>(metric.domain()));
132 ddc::Chunk new_metric_prod_alloc_(new_metric_prod_dom_, ddc::HostAllocator<double>());
133 sil::tensor::Tensor<

196// Apply metrics inplace (like g_mu_muprime*T^muprime^nu)
197template <class MetricIndex, class Indices1, class Indices2, class MetricType, class TensorType>
198relabelize_indices_of_t<TensorType, Indices2, Indices1>
199inplace_apply_metrics( // TODO avoid metricS by using concepts
200 TensorType tensor,
201 MetricType metric)
202{
203 sil::tensor::tensor_accessor_for_domain_t<metric_prod_domain_t<MetricIndex, Indices1, Indices2>>
204 metric_prod_accessor;
205 ddc::Chunk metric_prod_alloc(metric_prod_accessor.mem_domain(), ddc::HostAllocator<double>());
206 sil::tensor::Tensor<
207 double,
208 metric_prod_domain_t<MetricIndex, Indices1, Indices2>,
209 Kokkos::layout_right,
210 Kokkos::DefaultHostExecutionSpace::memory_space>
211 metric_prod(metric_prod_alloc);
212
213 fill_metric_prod<MetricIndex, Indices1, Indices2>(metric_prod, metric);
214
215 ddc::Chunk result_alloc(
216 relabelize_indices_in_domain<Indices2, Indices1>(tensor.domain()),
217 ddc::HostAllocator<double>());
218 relabelize_indices_of_t<TensorType, Indices2, Indices1> result(result_alloc);
219 tensor_prod(
220 result,
221 relabelize_indices_of<Indices2, typename detail::Primes<Indices1>::type>(metric_prod),
222 relabelize_indices_of<Indices2, typename detail::Primes<Indices2>::type>(tensor));
223 Kokkos::deep_copy(
224 tensor.allocation_kokkos_view(),
225 result.allocation_kokkos_view()); // We rely on Kokkos::deep_copy in place of ddc::parallel_deepcopy to avoid type verification of the type dimensions
226
227 return relabelize_indices_of<Indices2, Indices1>(tensor);
228}
