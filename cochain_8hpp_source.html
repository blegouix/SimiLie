167 KOKKOS_FUNCTION element_type integrate() noexcept
168 {
169 element_type out = 0;
170 for (auto i = m_values.begin(); i < m_values.end(); ++i) {
171 if constexpr (misc::Specialization<chain_type, Chain>) {
172 out += (chain_it(i)->negative() ? -1 : 1) * *i;
173 } else if (misc::Specialization<chain_type, LocalChain>) {
174 out += (m_chain.negative() ? -1 : 1) * *i; // always false
175 }
176 }
177 return out;
178 }
