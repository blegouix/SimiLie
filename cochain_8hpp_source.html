81 template <

88 KOKKOS_FUNCTION constexpr explicit Cochain(
89 chain_type& chain,
90 tensor::Tensor<
91 OElementType,
92 ddc::DiscreteDomain<Index>,
93 LayoutStridedPolicy,
94 MemorySpace> tensor) noexcept
95 : m_chain(chain)
96 , m_values("cochain_values", tensor.domain().size())
97 {
98 assert(m_values.size() == chain.size()
99 && "cochain constructor must get as much values as the chain contains simplices");
100 for (auto i = Kokkos::Experimental::begin(m_values);
101 i < Kokkos::Experimental::end(m_values);
102 ++i) {
103 *i = tensor.mem(ddc::DiscreteElement<Index>(
104 Kokkos::Experimental::distance(Kokkos::Experimental::begin(m_values), i)));
105 }

187 KOKKOS_FUNCTION element_type integrate() noexcept
188 {
189 element_type out = 0;
190 for (auto i = begin(); i < end(); ++i) {
191 if constexpr (misc::Specialization<chain_type, Chain>) {
192 out += ((*i).negative() ? -1 : 1) * (*i).value();
193 } else if (misc::Specialization<chain_type, LocalChain>) {
194 out += (m_chain.negative() ? -1 : 1) * (*i).value(); // always false
195 }
196 }
197 return out;
198 }
199
200 KOKKOS_FUNCTION element_type const integrate() const noexcept
201 {
202 element_type out = 0;
203 for (auto i = begin(); i < end(); ++i) {
204 if constexpr (misc::Specialization<chain_type, Chain>) {
205 out += ((*i).negative() ? -1 : 1) * (*i).value();
206 } else if (misc::Specialization<chain_type, LocalChain>) {
207 out += (m_chain.negative() ? -1 : 1) * (*i).value(); // always false
208 }
209 }
210 return out;

246 {

365template <misc::Specialization<Cochain> CochainType>
