88 KOKKOS_FUNCTION int check()
89 {
90 for (auto i = Kokkos::Experimental::begin(m_simplices);
91 i < Kokkos::Experimental::end(m_simplices) - 1;
92 ++i) {
93 for (auto j = i + 1; j < Kokkos::Experimental::end(m_simplices); ++j) {
94 if (*i == *j) {
95 return -1;
96 }

102 KOKKOS_FUNCTION void optimize()
103 {
104 auto i = Kokkos::Experimental::begin(m_simplices);
105 auto end = Kokkos::Experimental::end(m_simplices);
106 while (i < end - 1) {
107 auto k = i;
108 for (auto j = i + 1; k == i && j < end; ++j) {
109 if (*i == -*j) {
110 k = j;
111 }
112 }
113 if (k != i) {
114 Kokkos::Experimental::shift_left(
115 Kokkos::DefaultHostExecutionSpace(),
116 k,
117 Kokkos::Experimental::end(m_simplices),
118 1);
119 Kokkos::Experimental::shift_left(
120 Kokkos::DefaultHostExecutionSpace(),
121 i,
122 Kokkos::Experimental::end(m_simplices),
123 1);
124 Kokkos::resize(m_simplices, m_simplices.size() - 2);
125 end = Kokkos::Experimental::end(m_simplices);
126 } else {
127 i++;
128 }
129 }
130 assert(check() == 0 && "there are duplicate simplices in the chain");
131 }
