68 KOKKOS_FUNCTION int check()
69 {
70 for (auto i = m_simplices.begin(); i < m_simplices.end() - 1; ++i) {
71 for (auto j = i + 1; j < m_simplices.end(); ++j) {
72 if (*i == *j) {
73 return -1;
74 }
75 }
76 }
77 return 0;

80 KOKKOS_FUNCTION void optimize()
81 {
82 for (auto i = m_simplices.begin(); i < m_simplices.end() - 1; ++i) {
83 auto k = i;
84 for (auto j = i + 1; k == i && j < m_simplices.end(); ++j) {
85 if (*i == -*j) {
86 k = j;
87 }
88 }
89 if (k != i) {
90 m_simplices.erase(k);
91 m_simplices.erase(i--);
92 }
93 }
94 assert(check() == 0 && "there are duplicate simplices in the chain");
95 }
