121 KOKKOS_FUNCTION int check()
122 {
123 for (auto i = begin(); i < end() - 1; ++i) {
124 for (auto j = i + 1; j < end(); ++j) {

126 return -1;
127 }
128 }

133 KOKKOS_FUNCTION void optimize()
134 {
135 auto i = begin();
136 auto stop = end();
137 while (i < stop - 1) {
138 auto k = i;
139 for (auto j = i + 1; k == i && j < stop; ++j) {
140 if (*i == -*j) {
141 k = j;
142 }
143 }
144 if (k != i) {
145 misc::detail::shift_left(k, stop, 1);
146 misc::detail::shift_left(i, stop, 1);
147 m_size -= 2;
148 stop = end();
149 } else {
150 i++;
151 }
152 }
153 assert(check() == 0 && "there are duplicate simplices in the chain");
154 }
