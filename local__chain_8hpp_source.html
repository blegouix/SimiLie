215template <std::size_t K, misc::NotSpecialization<ddc::DiscreteDomain>... Tag>
216KOKKOS_FUNCTION constexpr LocalChain<Simplex<K, Tag...>> tangent_basis()
217{
218 std::array<std::ptrdiff_t, sizeof...(Tag)> permutation
219 = {0 * ddc::type_seq_rank_v<Tag, ddc::detail::TypeSeq<Tag...>>...};
220 for (auto i = permutation.begin(); i < permutation.begin() + K; ++i) {
221 *i = 1;
222 }
223 std::vector<ddc::DiscreteVector<Tag...>> basis {};
224 std::size_t i = 0;
225 do {
226 basis.push_back(ddc::DiscreteVector<Tag...>());
227 ddc::detail::array(basis[i++]) = permutation;
228 } while (std::prev_permutation(permutation.begin(), permutation.end()));
229
230 return LocalChain<Simplex<K, Tag...>>(basis);
231}
