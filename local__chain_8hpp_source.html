76 template <misc::NotSpecialization<ddc::DiscreteVector>... T>
77 KOKKOS_FUNCTION constexpr explicit LocalChain(vects_type allocation, T... simplex) noexcept
78 : m_vects(std::move(allocation))
79 , m_size(sizeof...(T))
80 {
81 std::size_t i = 0;
82 ((m_vects(i++) = misc::filled_struct<discrete_vector_type>() + simplex.discrete_vector()),
83 ...);
84 assert(check() == 0 && "there are duplicate simplices in the chain");
85 if constexpr (sizeof...(T) > 1) {
86 assert(misc::are_all_equal(simplex.discrete_element()...)
87 && "LocalChain must contain simplices with same origin (if not, use Chain)");
88 }
89 assert((!simplex.negative() && ...)
90 && "negative simplices are not supported in LocalChain");
91 }

93 KOKKOS_FUNCTION constexpr explicit LocalChain(
94 vects_type allocation,
95 simplices_type simplices,
96 std::size_t size) noexcept
97 : m_vects(std::move(allocation))
98 , m_size(size)
99 {
100 detail::extract_vects(m_vects, simplices)
101 assert(check() == 0 && "there are duplicate simplices in the chain");
102#if not defined NDEBUG
103 Kokkos::View<discrete_element_type*, memory_space> elems(simplices.size());
104 for (auto i = Kokkos::Experimental::begin(simplices);
105 i < Kokkos::Experimental::end(simplices);
106 ++i) {
107 elems[Kokkos::Experimental::distance(Kokkos::Experimental::begin(simplices), i)]
108 = i->discrete_element();
109 }
110 assert(misc::are_all_equal(elems)
111 && "LocalChain must contain simplices with same origin (if not, use Chain)");
112
113 /* Does not compile with nvcc, so we inline it
114 assert(misc::all_of(
115 Kokkos::Experimental::begin(simplices),
116 Kokkos::Experimental::end(simplices),
117 KOKKOS_LAMBDA(const std::size_t i) { return !simplices[i].negative(); })
118 && "LocalChain must contain simplices with same origin (if not, use Chain)");
119 */
120 bool flag = true;
121 for (auto i = Kokkos::Experimental::begin(simplices);
122 i < Kokkos::Experimental::end(simplices);
123 ++i) {
124 flag &= !i->negative();
125 }
126 assert(flag && "LocalChain must contain simplices with same origin (if not, use Chain)");
127#endif
128 }

369template <std::size_t K, misc::NotSpecialization<ddc::DiscreteDomain>... Tag>
370constexpr LocalChain<Simplex<K, Tag...>> tangent_basis()
371{
372 std::array<std::ptrdiff_t, sizeof...(Tag)> permutation
373 = {0 * ddc::type_seq_rank_v<Tag, ddc::detail::TypeSeq<Tag...>>...};
374 for (auto i = permutation.begin(); i < permutation.begin() + K; ++i) {
375 *i = 1;
376 }
377 Kokkos::View<ddc::DiscreteVector<Tag...>*, Kokkos::HostSpace>
378 basis("", misc::binomial_coefficient(sizeof...(Tag), K));
379 std::size_t i = 0;
380 do {
381 basis(i) = ddc::DiscreteVector<Tag...>();
382 ddc::detail::array(basis(i++)) = permutation;
383 } while (std::prev_permutation(permutation.begin(), permutation.end()));
384
385 return LocalChain<Simplex<K, Tag...>>(basis, basis.size());
386}
