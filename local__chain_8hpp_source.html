18template <class SimplexType>
19std::vector<typename SimplexType::vect_type> extract_vects(std::vector<SimplexType> simplices)
20{
21 std::vector<typename SimplexType::vect_type> vects;
22 for (auto& simplex : simplices) {
23 vects.push_back(simplex.discrete_vector());
24 }
25 return vects;
26}

76 : m_vects(detail::extract_vects(simplices))
77 {
78 assert(check() == 0 && "there are duplicate simplices in the chain");
79 std::function<bool()> check_common_elem = [&]() {
80 std::vector<elem_type> elems(simplices.size());
81 for (auto i = simplices.begin(); i < simplices.end(); ++i) {
82 elems[std::distance(simplices.begin(), i)] = i->discrete_element();
83 }
84 return misc::are_all_equal(elems);
85 };
86 assert(check_common_elem()
87 && "LocalChain must contain simplices with same origin (if not, use Chain)");
88 assert(std::
89 all_of(simplices.begin(),
90 simplices.end(),
91 [&](const std::size_t i) { return !simplices[i].negative(); })
92 && "LocalChain must contain simplices with same origin (if not, use Chain)");
