21template <class SimplexType, class MemorySpace>
22Kokkos::View<typename SimplexType::discrete_vector_type*, MemorySpace> extract_vects(
23 Kokkos::View<SimplexType*, MemorySpace> simplices)
24{
25 Kokkos::View<typename SimplexType::discrete_vector_type*, MemorySpace> vects(
26 simplices.extent(0));
27 for (auto i = Kokkos::Experimental::begin(simplices); i < Kokkos::Experimental::end(simplices);
28 ++i) {
29 vects[i] = i->discrete_vector();
30 }
31 return vects;
32}

67 template <misc::NotSpecialization<ddc::DiscreteVector>... T>
68 requires misc::are_all_same<T...>
69 KOKKOS_FUNCTION constexpr explicit LocalChain(T... simplex) noexcept
70 : m_vects("local_chain_vects", sizeof...(T))
71 {
72 int i = 0;
73 ((m_vects(i++) = simplex.discrete_vector()), ...);
74 assert(check() == 0 && "there are duplicate simplices in the chain");
75 assert(misc::are_all_equal(simplex.discrete_element()...)
76 && "LocalChain must contain simplices with same origin (if not, use Chain)");
77 assert((!simplex.negative() && ...)
78 && "negative simplices are not supported in LocalChain");
79 }

81 KOKKOS_FUNCTION explicit LocalChain(
82 simplices_type simplices) noexcept // Can be constexpr with C++23
83 : m_vects(detail::extract_vects(simplices))
84 {
85 assert(check() == 0 && "there are duplicate simplices in the chain");
86 std::function<bool()> check_common_elem = [&]() {
87 Kokkos::View<discrete_element_type*, memory_space> elems(simplices.size());
88 for (auto i = Kokkos::Experimental::begin(simplices);
89 i < Kokkos::Experimental::end(simplices);
90 ++i) {
91 elems[Kokkos::Experimental::distance(Kokkos::Experimental::begin(simplices), i)]
92 = i->discrete_element();
93 }
94 return misc::are_all_equal(elems);
95 };
96 assert(check_common_elem()
97 && "LocalChain must contain simplices with same origin (if not, use Chain)");
98 assert(std::
99 all_of(Kokkos::Experimental::begin(simplices),
100 Kokkos::Experimental::end(simplices),
101 [&](const std::size_t i) { return !simplices[i].negative(); })
102 && "LocalChain must contain simplices with same origin (if not, use Chain)");
103 }

252 KOKKOS_FUNCTION auto operator*(T t)
253 {
254 if (t == 1) {
255 return *this;
256 } else if (t == -1) {
257 assert(false && "negative simplices are unsupported in LocalChain");
258 } else {
259 assert(false && "chain must be multiplied by 1 or -1");

267 return false;

277template <std::size_t K, misc::NotSpecialization<ddc::DiscreteDomain>... Tag>
278KOKKOS_FUNCTION constexpr LocalChain<Simplex<K, Tag...>> tangent_basis()
279{
280 std::array<std::ptrdiff_t, sizeof...(Tag)> permutation
281 = {0 * ddc::type_seq_rank_v<Tag, ddc::detail::TypeSeq<Tag...>>...};
282 for (auto i = permutation.begin(); i < permutation.begin() + K; ++i) {
283 *i = 1;
284 }
285 Kokkos::View<ddc::DiscreteVector<Tag...>*, Kokkos::HostSpace>
286 basis("", misc::binomial_coefficient(sizeof...(Tag), K));
287 std::size_t i = 0;
288 do {
289 basis(i) = ddc::DiscreteVector<Tag...>();
290 ddc::detail::array(basis(i++)) = permutation;
291 } while (std::prev_permutation(permutation.begin(), permutation.end()));
292
293 return LocalChain<Simplex<K, Tag...>>(basis);
294}
