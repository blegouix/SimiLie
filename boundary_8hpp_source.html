41template <class SimplexType, misc::Specialization<Kokkos::View> AllocationType>
42KOKKOS_FUNCTION constexpr Chain<boundary_t<SimplexType>> generate_half_subchain(
43 AllocationType allocation,
44 typename SimplexType::discrete_element_type elem,
45 typename SimplexType::discrete_vector_type vect,
46 bool negative = false)
47{
48 auto array = ddc::detail::array(vect);
49 Chain<boundary_t<SimplexType>> chain(allocation);
50 auto id_dist = -1;
51 for (std::size_t i = 0; i < SimplexType::dimension(); ++i) {
52 auto array_ = array;
53 auto j = array_.begin() + id_dist + 1;
54 auto id = std::find_if(j, array_.end(), [](int k) { return k != 0; });
55 id_dist = std::distance(array_.begin(), id);
56 *id = 0;
57 typename SimplexType::discrete_vector_type vect_;
58 ddc::detail::array(vect_) = array_;
59 chain += boundary_t<SimplexType>(elem, vect_, (negative + i) % 2);
60 j = id + 1;
61 }
62 return chain;
63}

67template <misc::Specialization<Kokkos::View> AllocationType, class SimplexType>
68KOKKOS_FUNCTION Chain<boundary_t<SimplexType>> boundary(
69 AllocationType allocation,
70 SimplexType simplex)
71{
72 Chain<boundary_t<SimplexType>> chain(allocation);
73 detail::generate_half_subchain<SimplexType>(
74 Kokkos::
75 subview(allocation,
76 std::pair<std::size_t, std::size_t>(0, SimplexType::dimension())),
77 simplex.discrete_element(),
78 simplex.discrete_vector(),
79 SimplexType::dimension() % 2);
80 chain += SimplexType::dimension();
81 detail::generate_half_subchain<SimplexType>(
82 Kokkos::
83 subview(allocation,
84 std::pair<std::size_t, std::size_t>(
85 SimplexType::dimension(),
86 2 * SimplexType::dimension())),
87 simplex.discrete_element() + simplex.discrete_vector(),
88 -simplex.discrete_vector());
89 chain += SimplexType::dimension();
90 chain *= (SimplexType::dimension() % 2 ? 1 : -1) * (simplex.negative() ? -1 : 1);
91 return chain;
92}

94template <misc::Specialization<Kokkos::View> AllocationType, class SimplexType>
95KOKKOS_FUNCTION Chain<boundary_t<SimplexType>> boundary(
96 AllocationType allocation,
97 Chain<SimplexType> chain)
98{
99 Chain<boundary_t<SimplexType>> boundary_chain(allocation);
100 for (auto i = chain.begin(); i < chain.end(); ++i) {
101 std::size_t const distance = Kokkos::Experimental::distance(chain.begin(), i);
102 boundary(
103 Kokkos::
104 subview(allocation,
105 std::pair<std::size_t, std::size_t>(
106 2 * SimplexType::dimension() * distance,
107 2 * SimplexType::dimension() * (distance + 1))),
108 *i);
109 boundary_chain += 2 * SimplexType::dimension();
110 }
111 boundary_chain.optimize();
112 return boundary_chain;
113}
