41template <class SimplexType>
42KOKKOS_FUNCTION constexpr Chain<boundary_t<SimplexType>> generate_half_subchain(
43 typename SimplexType::discrete_element_type elem,
44 typename SimplexType::discrete_vector_type vect,
45 bool negative = false)
46{
47 auto array = ddc::detail::array(vect);
48 Chain<boundary_t<SimplexType>> chain;
49 auto id_dist = -1;
50 for (std::size_t i = 0; i < SimplexType::dimension(); ++i) {
51 auto array_ = array;
52 auto j = array_.begin() + id_dist + 1;
53 auto id = std::find_if(j, array_.end(), [](int k) { return k != 0; });
54 id_dist = std::distance(array_.begin(), id);
55 *id = 0;
56 typename SimplexType::discrete_vector_type vect_;
57 ddc::detail::array(vect_) = array_;
58 chain.push_back(boundary_t<SimplexType>(elem, vect_, (negative + i) % 2));
59 j = id + 1;
60 }
61 return chain;
62}

66template <class SimplexType>
67KOKKOS_FUNCTION Chain<boundary_t<SimplexType>> boundary(SimplexType simplex)
68{
69 return Chain<boundary_t<SimplexType>>(
70 detail::generate_half_subchain<SimplexType>(
71 simplex.discrete_element(),
72 simplex.discrete_vector(),
73 SimplexType::dimension() % 2)
74 + detail::generate_half_subchain<SimplexType>(
75 simplex.discrete_element() + simplex.discrete_vector(),
76 -simplex.discrete_vector()))
77 * (SimplexType::dimension() % 2 ? 1 : -1) * (simplex.negative() ? -1 : 1);
78}

80template <class SimplexType>
81KOKKOS_FUNCTION Chain<boundary_t<SimplexType>> boundary(Chain<SimplexType> chain)
82{
83 Chain<boundary_t<SimplexType>> boundary_chain;
84 for (auto& simplex : chain) {
85 Chain<boundary_t<SimplexType>> boundary_simplex = boundary(simplex);
86 for (auto& simplex_ : boundary_simplex) {
87 boundary_chain.push_back(simplex_);
88 }
89 }
90 boundary_chain.optimize();
91 return boundary_chain;
92}
